function renderHistorialCompleto(container, temario) { if (!temario || !temario.semanas || temario.semanas.length === 0) { container.innerHTML = `

No hay semanas disponibles en el temario.

`; return; } // Obtener la semana actual del estudiante const select = document.getElementById('selectorEstudianteEnc'); const estudianteId = select ? parseInt(select.value) : null; if (!estudianteId) { container.innerHTML = `

Selecciona un estudiante primero.

`; return; } const inscriptions = JSON.parse(localStorage.getItem('inscriptions') || '[]'); const estudiante = inscriptions.find(ins => ins.id == estudianteId); if (!estudiante) { container.innerHTML = `

No se encontró el estudiante.

`; return; } // Calcular semana actual const fechaIngreso = new Date(estudiante.academico?.fechaInicio); if (isNaN(fechaIngreso.getTime())) { container.innerHTML = `

Fecha de ingreso no válida.

`; return; } const hoy = new Date(); const diffDias = Math.floor((hoy - fechaIngreso) / (1000 * 60 * 60 * 24)); let semanaActual = Math.floor(diffDias / 7) + 1; if (semanaActual < 1) semanaActual = 1; if (semanaActual > 44) semanaActual = 44; // Filtrar semanas: solo las que ya pasaron o la actual (≤ semanaActual) const semanasOrdenadas = [...temario.semanas] .filter(s => s.numero <= semanaActual) .sort((a, b) => a.numero - b.numero); if (semanasOrdenadas.length === 0) { container.innerHTML = `

Aún no hay semanas disponibles. Tu primera clase comenzará en la semana 1.

`; return; } let html = `

📚 ${temario.instrumento} - Nivel ${temario.nivel}

Semana actual: ${semanaActual} de 44 (Mostrando ${semanasOrdenadas.length} semanas disponibles)

`; semanasOrdenadas.forEach(s => { const materialHtml = renderizarMaterial(s.material || 'No especificado'); // Determinar si es la semana actual o una semana pasada const esActual = (s.numero === semanaActual); const badge = esActual ? 'ACTUAL' : ''; html += `
Semana ${s.numero} ${s.tema || 'Sin tema'} ${badge}
Descripción:
${s.descripcion || 'Sin descripción'}
Material:
${materialHtml}
Ponderación: ${s.ponderacion || 0} pts
${esActual ? `
Esta es tu semana actual. ¡A practicar!
` : ''}
`; }); // Indicador de semanas futuras (si hay más allá de la actual) const semanasFuturas = temario.semanas.filter(s => s.numero > semanaActual); if (semanasFuturas.length > 0) { html += `
${semanasFuturas.length} semana(s) futuras aún no disponibles. Avanza en tu curso para desbloquearlas.
`; } html += `
`; container.innerHTML = html; }
